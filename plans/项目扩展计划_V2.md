**策略示例**:
```python
from backend.services.backtest import Strategy

class DualMAStrategy(Strategy):
    """双均线策略"""

    def __init__(self, fast=5, slow=20):
        super().__init__()
        self.fast = fast
        self.slow = slow

    def on_bar(self, bar):
        fast_ma = self.ma(bar.close, self.fast)
        slow_ma = self.ma(bar.close, self.slow)

        # 金叉买入
        if fast_ma > slow_ma and not self.position:
            self.buy(bar.symbol, bar.close)

        # 死叉卖出
        elif fast_ma < slow_ma and self.position:
            self.sell(bar.symbol, bar.close)
```
